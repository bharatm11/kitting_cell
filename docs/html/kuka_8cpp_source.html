200 KDL::Chain kuka::makeChain() {
201  // Define LWR chain_
202  KDL::Chain chain_;
203  // base
204  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::None),
205  KDL::Frame::DH_Craig1989(0, 0, 0.33989, 0)));
206  // joint 1
207  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
208  KDL::Frame::DH_Craig1989(0, -M_PI_2, 0, 0)));
209  // joint 2
210  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
211  KDL::Frame::DH_Craig1989(0, M_PI_2, 0.40011, 0)));
212  // joint 3
213  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
214  KDL::Frame::DH_Craig1989(0, M_PI_2, 0, 0)));
215  // joint 4
216  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
217  KDL::Frame::DH_Craig1989(0, -M_PI_2, 0.40003, 0)));
218  // joint 5
219  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
220  KDL::Frame::DH_Craig1989(0, -M_PI_2, 0, 0)));
221  // joint 6
222  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
223  KDL::Frame::DH_Craig1989(0, M_PI_2, 0, 0)));
224  // joint 7 (with flange adapter)
225  chain_.addSegment(KDL::Segment(KDL::Joint(KDL::Joint::RotZ),
226  KDL::Frame::DH_Craig1989(0, 0, 0.12597, 0)));
227  kuka::kinematicChain_ = chain_;
228  return chain_;
229 }

317  jointCmd.header.seq = 0;
318  jointCmd.header.stamp.sec = 0;

320  jointCmd.header.frame_id = kuka::id++;
321  // input joint coordinates

326 KDL::Frame kuka::evalKinematicsFK() {
327  ros::spinOnce();
328  // reinitialize chain.
329  KDL::Chain chain = kuka::makeChain();
330  // initialize solvers
331  KDL::ChainFkSolverPos_recursive fksolver =
332  KDL::ChainFkSolverPos_recursive(chain);
333  KDL::Frame cartPos;
334  // put joint angles in KDL format
335  for (int k = 0; k < numJoints_; ++k) {
336  kuka::jointPosKdl_(k) = kuka::jointsState_.position[k];
337  }
338  // perform FK
339  bool status = fksolver.JntToCart(kuka::jointPosKdl_, cartPos);
340  kuka::currCartpos_ = cartPos;
341  return cartPos;
342 }
343 // performs forwardr kinematics on the current joint angles
344 KDL::JntArray kuka::evalKinematicsIK(KDL::Frame cartpos) {
345  // read subscriber
346  ros::spinOnce();
347  // perform forward kinematics
348  kuka::evalKinematicsFK();
349  // make chain
350  KDL::Chain chain = kuka::makeChain();
351  // initiate solvers
352  KDL::ChainFkSolverPos_recursive fksolver =
353  KDL::ChainFkSolverPos_recursive(chain);
354  KDL::ChainIkSolverVel_pinv iksolverv = KDL::ChainIkSolverVel_pinv(chain);
355  KDL::ChainIkSolverPos_NR iksolver(chain, fksolver, iksolverv, 100, 1e-4);
356  // perform IK
357  int ret = iksolver.CartToJnt(kuka::jointPosKdl_, cartpos,
358  kuka::newJointPosKdl_);
359  return kuka::newJointPosKdl_;
360 }
361 // this function normalizes the points to robot's joint limits
362 trajectory_msgs::JointTrajectoryPoint kuka::normalizePoints(
363  KDL::JntArray joints) {
364  trajectory_msgs::JointTrajectoryPoint point;
365  // joints can move between -+: 172,120,172,120,172,120,170
366  // double joint_bounds[] = {3.002, 2.0944,3.002, 2.0944,3.002, 2.0944, 3.002};
367  for (int i = 0; i < kuka::numJoints_; ++i) {
368  while (joints(i) > M_PI) {
369  joints(i) -= 2*M_PI;
370  }
371  while (joints(i) < -M_PI) {
372  joints(i) += 2*M_PI;
373  }
374  point.positions.push_back(joints(i));
375  }
376  return point;
377 }
